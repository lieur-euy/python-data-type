# Strings
name = "uwuie"
print(name)
#output 
#uwuie


# Integers
age = 2
print(age)
#output 
#2

# Floats
height = 2.2
print(height)
#output 
#2.2

# Boolean
is_cool = True
print(is_cool)
#output 
#True

# Lists
music = ["rock", "pop", "kpop", "metal"]
print(music)
#output 
#['rock', 'pop', 'kpop', 'metal']

# Tuples
coordinates = (4.8, 8.2)
print(coordinates)
#output 
#(4.8, 8.2)


# Dictionaries
person = {
  "name": "uwuie.com",
  "age": 10,
  "height": 1.2
}
print(person)
#output 
#{'name': 'uwuie.com', 'age': 10, 'height': 1.2}
